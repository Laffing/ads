238 bool GetBroadcastMsg::loadFromLocalPath() {
239  uint32_t blockTime = this->getBlockTime();
240  if (!blockTime) {
241  return false;
242  }
243 
244  uint32_t path = blockTime-(blockTime%BLOCKSEC);
245  char filename[64];
246  sprintf(filename, "bro/%08X.bin", path);
247  std::ifstream file(filename, std::ifstream::binary | std::ifstream::in);
248  if (!file.is_open()) {
249  m_loadedFromLocal = false;
250  return false;
251  }
252 
253  struct stat sstat;
254  stat(filename, &sstat);
255  int remainData = sstat.st_size;
256  m_header.fileSize = remainData;
257  m_header.path = path;
258  m_loadedFromLocal = true;
259 
260  while (remainData > 0) {
261  GetBroadcastResponse response{};
262  file.read((char*)&response.info, sizeof(response.info));
263  remainData -= sizeof(response.info);
264 
265  char message[response.info.msg_length];
266  file.read((char*)message, response.info.msg_length);
267  remainData -= response.info.msg_length;
268 
269  file.read((char*)&response.data, sizeof(response.data));
270  remainData -= sizeof(response.data);
271 
272  m_response.push_back(std::make_pair(response, std::string(message, response.info.msg_length)));
273  }
274  file.close();
275 
276  return true;
277 }
