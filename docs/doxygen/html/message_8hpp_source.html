860  if(*data1==MSGTYPE_BLK) {

862  ELOG("DBL message bad BLK len\n");
863  return(-1);

879  int valid[2];
880  return(ed25519_sign_open_batch(m,mlen,pk,rs,2,valid));
881  }
882  }
883  if(*data1==MSGTYPE_MSG) {

895  ELOG("DBL bad message hash_tree_fast 1 failed\n");
896  return(-1);
897  }

899  ed25519_sign_open2(dblmsha,32,sigh1,32,svpk,data1+4) ||
900  ed25519_sign_open2(dblmsha,32,sigh2,32,svpk,data2+4));
901  }
902  ELOG("DBL message illegal type\n");
903  return(-1);
904  }
