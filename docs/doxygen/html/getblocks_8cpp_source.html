310 bool GetBlocks::validateLastBlockUsingFirstKeys() {
311  std::map<uint16_t, hash_s> vipkeys;
312  for(uint32_t i=0; i<m_firstKeysLen; i+=2+32) {
313  char* firstkeys = m_firstVipKeys.getVipKeys();
314  uint16_t svid=*((uint16_t*)(&firstkeys[i]));
315  vipkeys[svid]=*((hash_s*)(&firstkeys[i+2]));
316  }
317 
318  int vok=0;
319 
320  for(const auto& s : m_signatures) {
321  auto it=vipkeys.find(s.svid);
322  if(it==vipkeys.end()) {
323  char hash[65];
324  hash[64]='\0';
325  ed25519_key2text(hash, s.signature, 32);
326  ELOG("ERROR vipkey (%d) not found %s\n", s.svid, hash);
327  continue;
328  }
329 
330  if(!ed25519_sign_open((const unsigned char*)(&m_receivedHeaders[m_numOfBlocks-1]), sizeof(header_t)-4, it->second.hash, s.signature)) {
331  vok++;
332  } else {
333  char hash[65];
334  hash[64]='\0';
335  ed25519_key2text(hash, s.signature, 32);
336  ELOG("ERROR vipkey (%d) failed %s\n", s.svid, hash);
337  }
338  }
339 
340  if(2*vok < (int)vipkeys.size()) {
341  ELOG("ERROR failed to confirm enough signature for header %08X (2*%d<%d)\n",
342  m_receivedHeaders[m_numOfBlocks-1].now, vok, (int)vipkeys.size());
343  return false;
344  } else {
345  ELOG("CONFIRMED enough signature for header %08X (2*%d>=%d)\n",
346  m_receivedHeaders[m_numOfBlocks-1].now, vok, (int)vipkeys.size());
347  }
348  return true;
