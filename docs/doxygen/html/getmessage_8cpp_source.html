100  }
101 
102  if (m_responseError) {
103  return true;
104  }
105 
106  if (!netClient.readData((int32_t*)&m_responseBlock, sizeof(uint32_t))) {
107  ELOG("GetMessage reading message block\n");
108  return false;
109  }
110 
111  // msgTypeLength contains type (1B) and length (3B)
112  uint32_t msgTypeLength = 0;
113  if (!netClient.readData((int32_t*)&msgTypeLength, sizeof(msgTypeLength))) {
114  ELOG("GetMessage reading length error\n");
115  return false;
116  }
117  uint32_t msgLength = msgTypeLength >> 8;
118  if (!msgTypeLength || msgLength < message::data_offset) {
119  m_responseError = ErrorCodes::Code::eBadLength;
120  return true;
121  }
122  uint8_t msgType = msgTypeLength & 0xFF;
123  if (msgType != MSGTYPE_MSG) {
124  m_responseError = ErrorCodes::Code::eIncorrectTransaction;
125  return true;
126  }
127 
128  char buffer[msgLength];
129  memcpy(buffer, &msgTypeLength, 4);
130  if (!netClient.readData(buffer + 4, msgLength - 4)) {
131  ELOG("GetMessage reading error\n");
132  return false;
133  }
134 
135  m_responseMsg.reset(new message(msgLength, (uint8_t*)buffer));
136 
137  return true;
