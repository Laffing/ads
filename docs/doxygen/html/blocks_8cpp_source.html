105  }

107  char previousSnapshotPath[64];
108  char backupFilePath[64];
109  unsigned int usert_size = sizeof(user_t);
110 
111  // for each node
112  for (uint16_t bank = 1; bank < nodes; ++bank)
113  {
114  // get undo files, youngest at begin
115  std::vector<std::shared_ptr<Helper::FileWrapper>> undoFiles;
116  std::shared_ptr<Helper::FileWrapper> undo_file;

125  }
126  }
127 
128  // load previous snapshot or usr/* for initial run
129  uint32_t users = get_users_count(bank);
130  Helper::FileName::getUndo(previousSnapshotPath, (block_path - (BLOCKDIV*BLOCKSEC)), bank);
131  if (!boost::filesystem::exists(previousSnapshotPath)) {
132  Helper::FileName::getUsr(previousSnapshotPath, bank);
133  }
134 
135  Helper::FileWrapper last_snapshot(previousSnapshotPath, O_RDONLY, false);
136  if (!last_snapshot.isOpen()) {
137  return;
138  }
139 
140  user_t u;

143 
144  int current_snapshot = open(backupFilePath, O_WRONLY | O_CREAT, 0644);
145  if (current_snapshot < 0) {
146  return;
147  }
148 
149  // for each user
