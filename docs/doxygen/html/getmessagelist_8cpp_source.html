154 void GetMessageList::toJson(boost::property_tree::ptree& ptree) {

166  ed25519_key2text(hash, m_responseTxnHash,32);
167  ptree.put("msghash",hash);
168  ptree.put("message_count",m_responseMessageList.size());
169  hashtree htree;
170  boost::property_tree::ptree messages;
171  for (auto &it : m_responseMessageList) {
172  boost::property_tree::ptree msghash;
173  ed25519_key2text(hash, it.hash,32);
174  msghash.put_value(Helper::print_msg_pack_id(it.node_id, it.node_msid));
175  htree.update(it.hash);
176  messages.push_back(std::make_pair("",msghash));
177  }
178  uint8_t fullHash[SHA256_DIGEST_LENGTH];
179  htree.finish(fullHash);
180 
181 
182  if(memcmp(m_responseTxnHash, fullHash, SHA256_DIGEST_LENGTH)) {
183  ed25519_key2text(hash, fullHash, SHA256_DIGEST_LENGTH);
184  ptree.put("msghash_calculated",hash);
185  ptree.put("confirmed","no");

187  ptree.put("confirmed","yes");
