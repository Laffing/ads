63  uint16_t crc16(const uint8_t* data_p, uint8_t length) {
64  uint8_t x;
65  uint16_t crc = 0x1D0F; //differet initial checksum !!!
66 
67  while(length--) {
68  x = crc >> 8 ^ *data_p++;
69  x ^= x>>4;
70  crc = (crc << 8) ^ ((uint16_t)(x << 12)) ^ ((uint16_t)(x <<5)) ^ ((uint16_t)x);
71  }
72  return crc;
73  }
