88  str_amount.insert(str_amount.length(), AMOUNT_DECIMALS, '0');

129 int check_csum(user_t& u,uint16_t peer,uint32_t uid) {

132  SHA256_Init(&sha256);
133  SHA256_Update(&sha256,&u,sizeof(user_t)-4*sizeof(uint64_t));
